  mutate(pct_dif_high = (income_high_dif / income_high)) |>
  mutate(pct_dif_missing = (income_missing_dif / income_missing)) |>
  mutate(subject = fct_recode(subject,
                              "Math" = "math",
                              "Spelling" = "spelling",
                              "Reading" = "reading"))

write.csv(pre_covid, "../../data/pre_covid.csv")
